### 三、核心功能实现

#### 1. 笔记服务

```java
@Service
public class NoteService {
    
    @Autowired
    private MongoTemplate mongoTemplate;
    
    // 创建笔记
    public Note createNote(Note note) {
        note.setCreateTime(LocalDateTime.now());
        note.setUpdateTime(LocalDateTime.now());
        note.setLikeCount(0);
        note.setCommentCount(0);
        note.setShareCount(0);
        return mongoTemplate.insert(note);
    }
    
    // 获取笔记详情
    public Note getNoteById(String id) {
        Query query = new Query(Criteria.where("id").is(id));
        return mongoTemplate.findOne(query, Note.class);
    }
    
    // 更新笔记
    public Note updateNote(Note note) {
        note.setUpdateTime(LocalDateTime.now());
        Update update = new Update();
        BeanUtils.describe(note).forEach((key, value) -> {
            if (value != null && !"id".equals(key)) {
                update.set(key, value);
            }
        });
        mongoTemplate.updateFirst(
            Query.query(Criteria.where("id").is(note.getId())),
            update,
            Note.class
        );
        return note;
    }
    
    // 删除笔记
    public void deleteNote(String id) {
        mongoTemplate.remove(Query.query(Criteria.where("id").is(id)), Note.class);
    }
    
    // 获取用户笔记列表
    public List<Note> getNotesByUserId(Long userId, int page, int size) {
        Query query = new Query(Criteria.where("userId").is(userId))
            .with(Sort.by(Sort.Direction.DESC, "createTime"))
            .skip((long) (page - 1) * size)
            .limit(size);
        return mongoTemplate.find(query, Note.class);
    }
    
    // 附近的笔记（地理位置查询）
    public List<Note> getNearbyNotes(double longitude, double latitude, double distance) {
        Query query = new Query(
            Criteria.where("location").nearSphere(new Point(longitude, latitude))
                .maxDistance(distance / 6371) // 距离单位：公里
        );
        return mongoTemplate.find(query, Note.class);
    }
}
```

#### 2. 评论服务

```java
@Service
public class CommentService {
    
    @Autowired
    private MongoTemplate mongoTemplate;
    
    // 添加评论
    public Comment addComment(Comment comment) {
        comment.setCreateTime(LocalDateTime.now());
        comment.setLikeCount(0);
        Comment savedComment = mongoTemplate.insert(comment);
        
        // 更新笔记评论数
        Update update = new Update().inc("commentCount", 1);
        mongoTemplate.updateFirst(
            Query.query(Criteria.where("id").is(comment.getNoteId())),
            update,
            Note.class
        );
        
        return savedComment;
    }
    
    // 获取笔记评论列表
    public List<Comment> getCommentsByNoteId(String noteId, int page, int size) {
        Query query = new Query(Criteria.where("noteId").is(noteId))
            .with(Sort.by(Sort.Direction.DESC, "createTime"))
            .skip((long) (page - 1) * size)
            .limit(size);
        return mongoTemplate.find(query, Comment.class);
    }
    
    // 添加回复
    public void addReply(String commentId, Reply reply) {
        reply.setCreateTime(LocalDateTime.now());
        Update update = new Update().push("replies", reply);
        mongoTemplate.updateFirst(
            Query.query(Criteria.where("id").is(commentId)),
            update,
            Comment.class
        );
    }
}
```


### 四、MongoDB 高级应用

#### 1. 聚合框架实现热门笔记

```java
// 获取热门笔记（点赞数+评论数）
public List<Note> getHotNotes(int page, int size) {
    Aggregation aggregation = Aggregation.newAggregation(
        // 过滤条件
        Aggregation.match(Criteria.where("isPublic").is(true)),
        
        // 排序（点赞数+评论数）
        Aggregation.sort(Sort.by(Sort.Direction.DESC, "likeCount", "commentCount")),
        
        // 分页
        Aggregation.skip((long) (page - 1) * size),
        Aggregation.limit(size)
    );
    
    return mongoTemplate.aggregate(aggregation, "notes", Note.class).getMappedResults();
}
```

#### 2. 全文搜索功能

```java
// 创建全文索引
@PostConstruct
public void createTextIndex() {
    mongoTemplate.indexOps(Note.class).ensureIndex(
        new Index().on("title", Sort.Direction.ASC).on("content", Sort.Direction.ASC).text()
    );
}

// 全文搜索笔记
public List<Note> searchNotes(String keyword, int page, int size) {
    Query query = new Query(
        TextCriteria.forDefaultLanguage().matchingAny(keyword)
    ).with(Sort.by(TextScore.score()).descending())
     .skip((long) (page - 1) * size)
     .limit(size);
    
    return mongoTemplate.find(query, Note.class);
}
```

#### 3. 批量操作

```java
// 批量更新笔记点赞数
public void batchUpdateLikeCount(Map<String, Integer> noteLikeCounts) {
    List<WriteModel<Note>> writes = new ArrayList<>();
    
    noteLikeCounts.forEach((noteId, count) -> {
        Update update = new Update().set("likeCount", count);
        writes.add(new UpdateOneModel<>(
            new Document("_id", noteId),
            update
        ));
    });
    
    mongoTemplate.bulkOps(BulkOperations.Type.UNORDERED, Note.class).execute(writes);
}
```

#### 1. 热点数据缓存

```java
// 获取笔记详情（先查Redis，再查MongoDB）
public Note getNoteWithCache(String id) {
    String key = "note:" + id;
    
    // 从Redis获取
    Note note = (Note) redisTemplate.opsForValue().get(key);
    
    if (note == null) {
        // 从MongoDB获取
        note = noteService.getNoteById(id);
        
        if (note != null) {
            // 存入Redis缓存
            redisTemplate.opsForValue().set(key, note, 1, TimeUnit.HOURS);
        }
    }
    
    return note;
}
```

```java
// 为笔记集合创建复合索引
@PostConstruct
public void createIndexes() {
    // 为热门查询创建复合索引
    mongoTemplate.indexOps(Note.class).ensureIndex(
        new Index().on("createTime", Sort.Direction.DESC)
    );
    
    // 为用户查询创建索引
    mongoTemplate.indexOps(Note.class).ensureIndex(
        new Index().on("userId", Sort.Direction.ASC).on("createTime", Sort.Direction.DESC)
    );
    
    // 为地理位置查询创建索引
    mongoTemplate.indexOps(Note.class).ensureIndex(
        new GeospatialIndex("location")
    );
}
```
